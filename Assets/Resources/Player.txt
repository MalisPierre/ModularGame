function Update()
	if (Framework.CheckInputDown(Framework.CreateKeyCode("Space"))) then
		PlayerPos = Framework.GetPosition()
		Framework.IDebug("Position ==> [" .. PlayerPos.x .. "][" .. PlayerPos.y .. "][" .. PlayerPos.z .. "]")
	end
	if (Framework.CheckInput(Framework.CreateKeyCode("Z"))) then
		Framework.IDebug("Moving Forward")
		Framework.MoveForward(1.0)
	elseif (Framework.CheckInput(Framework.CreateKeyCode("S"))) then
		Framework.IDebug("Moving Backward")
		Framework.MoveForward(-1.0)
	end
	if (Framework.CheckInput(Framework.CreateKeyCode("D"))) then
		Framework.IDebug("Moving Right")
		Framework.MoveRight(1.0)
	elseif (Framework.CheckInput(Framework.CreateKeyCode("Q"))) then
		Framework.IDebug("Moving Left")
		Framework.MoveRight(-1.0)
	end
end

function Init()
	Framework.IDebug("TEST INIT")
end
